#include<bits/stdc++.h>
using namespace std;

#define int long long

const int N=5e5+5;

int n,m;
vector<int> G[N];
int dfn[N],low[N],tim,sz[N],ans[N];

void tarjan(int u)
{
    dfn[u]=low[u]=++tim;
    sz[u]=1;
    int siz=0;
    for(int v:G[u]){
        if(!dfn[v]){
            tarjan(v);
            sz[u]+=sz[v];
            low[u]=min(low[u],low[v]);
            if(low[v]>=dfn[u]){
                ans[u]+=sz[v]*siz;
                siz+=sz[v];
            }
        }else
            low[u]=min(low[u],dfn[v]);
    }
    ans[u]+=(n-siz-1)*siz+n-1;
}

signed main()
{
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    cin>>n>>m;
    for(int i=1,u,v;i<=m;i++){
        cin>>u>>v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    tarjan(1);
    for(int i=1;i<=n;i++)
        cout<<ans[i]*2<<"\n";
    return 0;
}
