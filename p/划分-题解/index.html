const int P=998244353,N=2e3+10;

int n,ans;

int popcount(int x)
{
    int cnt=0;
    for(int i=63;~i;i--)
        cnt+=((x>>i)&1);
    return cnt;
}

signed main()
{
    ans=1;
    scanf("%lld",&n);
    for(int i=1;i<=n;i++){
        if(popcount(i)&1)
            ans=ans*i%P;
    }
    printf("%lld\n",ans);
    return 0;
}
