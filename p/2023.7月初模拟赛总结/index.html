            else
                op = 0;

                op = 0;
            else
                op = 1;

            if (a[k].y > a[j].y)
                op = 0;
            else
                op = 1;

inline ll read() {
    ll x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') x = (x << 3) + (x << 1) + (ch ^ 48), ch = getchar();
    return x * f;
}

inline void write(ll x) {
    if (x < 0) {
        putchar('-'), write(-x);
        return;
    }
    if (x > 9)
        write(x / 10), putchar(x % 10 + 48);
    else
        putchar(x + 48);
}

void getprime(ll n) {
    ans[1] = 1;
    s[1] = 1;
    for (ll i = 2; i <= n; ++i) {
        if (!ispr[i]) {
            pri[++pinum] = i;
            if (i % 4)
                s[i % 4] = i;
        }
        for (ll j = 1; j <= pinum; ++j) {
            if (i * pri[j] > n)

            ispr[i * pri[j]] = 1;
            if (i % pri[j] == 0)
                break;
        }
        if (i % 4 == 0)
            ans[i] = i / 4 + 1;
        else
            ans[i] = (i - s[i % 4]) / 4 + 1;
    }
}
