#include<bits/stdc++.h>
using namespace std;

#define int long long

const int P=998244353,phi=998244352,N=1e3+5;

string s;
int n,m;
int res[6]={0,0,48,5120,1376256,209715199};
int ans;

int qpow(int a,int b)
{
    int c=1;
    for(;b;b>>=1,a=a*a%P)
        if(b&1)
            c=c*a%P;
    return c;
}

signed main()
{
    cin>>s;
    int len=(int)s.size();
    for(int i=0;i<len;i++)
        n=(n*10%P+(s[i]^48))%P;
    for(int i=0;i<len;i++)
        m=(m*10%phi+(s[i]^48))%phi;
    if(n<=5)
        return printf("%lld\n",res[n]),0;
    int t4=qpow(2,phi-4)*12%P;
    int t6=qpow(2,phi-6)*80%P;
    int t9=qpow(2,phi-9)*1024%P;
    (ans+=t4*4)%=P;
    (ans+=t6*4%P*(n-2)%P)%=P;
    (ans+=t9*qpow(n-2,2)%P)%=P;
    ans=ans*qpow(2,m*m%phi)%P;
    printf("%lld\n",ans);
    return 0;
}
