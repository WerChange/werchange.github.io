#include<bits/stdc++.h>
using namespace std;

#define int long long
#define pii pair<int,int>
#define ft first
#define sd second

const int N=5e6+5;

int n,m;
int t[N],b[N];
pii a[N];
int tr[N];
int tmp[N],tot;
map<int,int> Hs;

void add(int x,int k) {for(;x<=tot;x+=x&-x)tr[x]=max(tr[x],k);}

int query(int x) {int c=0;for(;x;x-=x&-x)c=max(c,tr[x]);return c;}

bool cmp(pii x,pii y)
{
    if(x.ft!=y.ft)
        return x.ft<y.ft;
    return x.sd>y.sd;
}

signed main()
{
    scanf("%lld%lld",&n,&m);
    int Max=0;
    for(int i=1;i<=m;i++)
        scanf("%lld",&t[i]),Max=max(Max,t[i]);
    for(int i=1;i<=m;i++)
    {
        scanf("%lld",&b[i]);
        tmp[++tot]=b[i]-(Max-t[i]);
        tmp[++tot]=b[i]+Max-t[i];
    }
    sort(tmp+1,tmp+tot+1);
    for(int i=1;i<=tot;i++)
        Hs[tmp[i]]=i;
    for(int i=1;i<=m;i++)
    {
        a[i].ft=Hs[b[i]-(Max-t[i])];
        a[i].sd=Hs[b[i]+Max-t[i]];
    }
    sort(a+1,a+m+1,cmp);
    for(int i=1;i<=m;i++)
        add(a[i].sd,query(a[i].sd-1)+1);
    printf("%lld\n",query(tot));
    return 0;
}
