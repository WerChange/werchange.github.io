#include<bits/stdc++.h>
using namespace std;

#define int long long

const int P=1e9+7,N=6e3+5;

int n,m;
int fac[N],inv[N],pow2[N];
int f[N][N];

int qpow(int a,int b=P-2)
{
    int c=1;
    for(;b;b>>=1,a=a*a%P)
        if(b&1)
            c=c*a%P;
    return c;
}

void init()
{
    fac[0]=1;
    for(int i=1;i<=N-5;i++)
        fac[i]=fac[i-1]*i%P;
    inv[N-5]=qpow(fac[N-5]);
    for(int i=N-5;i;i--)
        inv[i-1]=inv[i]*i%P;
    int inv2=qpow(2);
    pow2[0]=1;
    for(int i=1;i<=N-5;i++)
        pow2[i]=pow2[i-1]*inv2%P;
}

int C(int n,int m)
{
    if(n<m) return 0;
    return fac[n]*inv[m]%P*inv[n-m]%P;
}

int work(int i)
{
    int res=0;
    for(int j=0;j<=n-i;j++)
        (res+=pow2[j]*C(n-i,j)%P*f[n-i+j][m-i]%P)%=P;
    return res;
}

signed main()
{
    scanf("%lld%lld",&n,&m);
    init();
    f[0][0]=1;
    for(int i=1;i<=2*n;i++)
        for(int j=1;j<=m;j++){
            f[i][j]=f[i-1][j-1]*i%P;
            if(i>=2)
                (f[i][j]+=f[i-2][j-1]*C(i,2)%P)%=P;
        }
    int ans=0;
    for(int i=0;i<=n&&i<=m;i++)
        (ans+=C(n,i)*C(m,i)%P*fac[i]%P*((i&1)?-1:1)*pow2[i]%P*work(i)%P)%=P;
    ans=(ans%P+P)%P;
    printf("%lld\n",ans);
    return 0;
}
