int n;
int a[MAXN], b[MAXN], c[MAXN];

struct XXJ {

        memset(d, 0, sizeof(d));
        memset(id, 0, sizeof(id));
        tot = 0;
    }
    bool Count(int x) {
        for (int i = 59; ~i; i--)
            if (x >> i & 1) {
                if (d[i] == 0)
                    return 0;
                x ^= d[i];
            }
        return 1;
    }
    int Insert(int x) {
        int res = 0;
        for (int i = 60; ~i; i--)
            if (x >> i & 1) {
                if (d[i] == 0) {
                    d[i] = x;
                    id[i] = tot++;
                    return res | (1ll << id[i]);
                }
                res |= (1ll << id[i]);
                x ^= d[i];
            }
        return res;
    }
} X;

void work(int x) {
    ans.push_back(x);
    for (int i = 1; i <= x; i++) {
        b[i] ^= b[i - 1];
        c[i] ^= c[i - 1];
    }
}
