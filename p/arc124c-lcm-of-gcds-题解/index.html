#include <bits/stdc++.h>
using namespace std;

#define int long long

const int MAXN = 50 + 5;

int n;
int a[MAXN], b[MAXN];

int solve() {
    int x = a[1], y = b[1];
    for (int i = 2; i <= n; i++) {
        int nx = __gcd(x, a[i]);
        int ny = __gcd(y, b[i]);
        int mx = __gcd(x, b[i]);
        int my = __gcd(y, a[i]);
        if (nx * ny > mx * my)
            x = nx, y = ny;
        else
            x = mx, y = my;
    }
    return x * y / gcd(x, y);
}

signed main() {
    scanf("%lld", &n);
    for (int i = 1; i <= n; i++) scanf("%lld%lld", &a[i], &b[i]);
    int ans1 = solve();
    for (int i = 1, j = n; i <= j; i++, j--) swap(a[i], a[j]), swap(b[i], b[j]);
    int ans2 = solve();
    printf("%lld\n", max(ans1, ans2));
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

#define int long long

const int MAXN = 50 + 5;

int n;
int a[MAXN], b[MAXN];
struct node {
    int x, ga, gb;
    node(int a = 0, int b = 0, int c = 0) {
        x = a;
        ga = b;
        gb = c;
    }
    bool operator<(const node &T) const {
        if (x != T.x)
            return x < T.x;
        if (ga != T.ga)
            return ga < T.ga;
        return gb < T.gb;
    }
};
int ans;
map<node, bool> vis;

int lcm(int a, int b) { return a * b / __gcd(a, b); }

void dfs(int x, int ga, int gb) {
    if (vis[node(x, ga, gb)])
        return;
    vis[node(x, ga, gb)] = 1;
    if (x == n + 1) {
        ans = max(ans, lcm(ga, gb));
        return;
    }
    dfs(x + 1, __gcd(ga, a[x]), __gcd(gb, b[x]));
    dfs(x + 1, __gcd(ga, b[x]), __gcd(gb, a[x]));
}

signed main() {
    scanf("%lld", &n);
    for (int i = 1; i <= n; i++) scanf("%lld%lld", &a[i], &b[i]);
    dfs(2, a[1], b[1]);
    printf("%lld\n", ans);
    return 0;
}
