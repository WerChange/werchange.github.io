#include <bits/stdc++.h>
using namespace std;

#define int long long

const int MAXN = 2e5 + 5;

int n;
int a[MAXN];
int d[MAXN];
int f[MAXN][2];

bool check(int md) {
    memset(f, -1, sizeof(f));
    f[1][1] = md;
    for (int i = 2; i <= n; i++) {
        if (f[i - 1][1] != -1) {
            int k = f[i - 1][1];
            if (k - d[i] / 2 >= 0) {
                f[i][0] = max(f[i][0], md - d[i] / 2);
                f[i][1] = max(f[i][1], k - d[i] / 2);
            }
        }
        if (f[i - 1][0] != -1) {
            int k = f[i - 1][0];
            if (k - d[i] / 2 >= 0) {
                f[i][0] = max(f[i][0], k - d[i] / 2);
                f[i][1] = max(f[i][1], k - d[i] / 2);
            }
        }
    }
    return f[n][0] != -1 || f[n][1] != -1;
}

signed main() {
    scanf("%lld", &n);
    for (int i = 1; i <= n; i++) scanf("%lld", &a[i]);
    for (int i = 2; i <= n; i++) d[i] = a[i] - a[i - 1];
    int l = 1, r = 1e9, mid, ans = -1;
    while (l <= r) {
        mid = l + r >> 1;
        if (check(mid))
            r = mid - 1, ans = mid;
        else
            l = mid + 1;
    }
    printf("%lld\n", ans);
    return 0;
}
